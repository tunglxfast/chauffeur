DROP TABLE IF EXISTS entry;
DROP TABLE IF EXISTS info_entry;

CREATE TABLE info_entry (
    id IDENTITY PRIMARY KEY,
    code VARCHAR(50),
    title VARCHAR(255),
    content TEXT,
    image_url VARCHAR(255)
);

CREATE TABLE entry (
    id IDENTITY PRIMARY KEY,
    title VARCHAR(255),
    description TEXT,
    image_url VARCHAR(255),
    published_date TIMESTAMP
);
